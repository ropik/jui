The following is code for creatinga calendar object. The format used when you click on the title and the date can be configured using options. 

jui.ready([ "ui.datepicker" ], function(datepicker) {
	datepicker1 = datepicker("#datepicker1", {
	    titleFormat: "yyyy MM",
	    format: "yyyy/MM/dd",
	    event: {
	       select: function(date, e) {
	           alert(date);
	       },
	       prev: function(e) {
	           alert("prev");
	       },
	       next: function(e) {
	           alert("next");
	       }
	    }
	});
});

Monthly & Yearly

The datepicker component allows the user to choose a month and a year.

The components of the calendar are separated into templates for setting a language. 

<script data-jui="#datepicker2" data-tpl="dates" type="text/template">
    <tr>
        <! for(var i = 0; i < dates.length; i++) { !>
        <td><!= dates[i] !>month</td>
        <! } !>
    </tr>
</script>

<script data-jui="#datepicker3" data-tpl="dates" type="text/template">
    <tr>
        <! for(var i = 0; i < dates.length; i++) { !>
        <td><!= dates[i] !>year</td>
        <! } !>
    </tr>
</script>

To change the calender type, use the type option. 

datepicker2 = datepicker("#datepicker2", {
	type: "monthly",
    titleFormat: "yyyy",
    format: "yyyy/MM",
    event: {
    	select: function(date, e) {
    		alert(date);
    	}
    }
});

datepicker3 = datepicker("#datepicker3", {
	type: "yearly",
    titleFormat: "Select a year",
    format: "yyyy",
    event: {
    	select: function(date, e) {
    		alert(date);
    	}
    }
});
