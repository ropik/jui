    Tab1

    A tab list can also be configured dynamically, and the related markup omitted. 

    <ul id="tab_2" class="tab tab-top"></ul>
<div id="tab_2_contents" style="background: #dcdcdc;">Tab1</div>

    It is possible to set the initial value of a tab list using the nodes option.
    In addition, it is possible to use a function to control the tab dynamically, such as the tab2Run function. 

    tab_2 = tab("#tab_2", {
	nodes: [
		{ text: "Tab1" },
		{ text: "Tab2" },
		{ text: "Tab3" }
	],
	event: {
		change: function(data) {
			$("#tab_2_contents").html(data.text);
		}
	},
	tpl: {
		node: "<li><a href='#'><!= text !></a></li>"
	}
});

function tab2Run(type) {
	switch(type) {
		case 1:
			tab_2_count += 1;
			tab_2.append({ text: "Tab" + tab_2_count });
			
			alert("Adds a tab to the last position.");
			break;
		case 2:
			tab_2_count += 1;
			tab_2.prepend({ text: "Tab" + tab_2_count });
			
			alert("Adds a tab to the first position.");
			break;
		case 3:
			tab_2_count += 1;
			tab_2.insert(2, { text: "Tab" + tab_2_count });

			alert("Adds to the Tab No. 2 position.");
			break;
		case 4:
			tab_2.remove(0);

			alert("Deletes Tab No. 0.");
			break;
		case 5:
			tab_2.move(0, 2);
			
			alert("Changes the position of Tab No. 0 to that of Tab No. 2.");
			break;
	}
}
