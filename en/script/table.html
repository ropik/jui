<table id="table_1" class="table table-classic">

			<th>Name</th>
			<th>Age</th>

Assign the selector of a table to the data-jui property value, and assign the data-tpl property value to a row. 

<script data-jui="#table_1" data-tpl="row" type="text/template">
	<tr>
		<td><!= name !></td>
		<td><!= age !></td>
		<td><!= location !></td>
	</tr>
</script>

		<input type="text" class="input input-small" value="<!= age !>" />
	</div>
	<div class="group">
		<label class="text text-small">Location</label>		
		<input type="text" class="input input-small" value="<!= location !>" />
	</div>
	<div class="group">
		<a href="javascript:table2Submit(<!= row.index !>);" class="btn btn-black btn-small">Submit</a>
		<a href="javascript:table_2.remove(<!= row.index !>); table_2.hideExpand();" class="btn btn-black btn-small">Delete</a>
		<a href="jabascript:table_2.hideExpand();" class="btn btn-black btn-small">Cancel</a>
	</div>
</script>

To use an expansion area, options must be added to the script. 

table_2 = table("#table_2", {
	data: table_data,
	expand: true,
	animate: true
});

table_2.on("expand", function(row, e) {
	$(row.list[0]).html("<i class='icon-right'></i>");
});
table_2.on("expandend", function(row, e) {
	$(row.list[0]).html("<i class='icon-left'></i>");
});

The code shown below is the row correction code that runs when the submit button on the expansion area is clicked. 

function table2Submit(index) {
	var name = $("#table_2_name").val(),
		age = $("#table_2_age").val(),
		location = $("#table_2_location").val();
		
	table_2.update(index, { name: name, age: age, location: location });
	table_2.hideExpand();
}

table_5.on("sort", function(column, e) {

		"desc": "icon-arrow1 icon-white",
		"asc": "icon-arrow3 icon-white"

Edit Row (+Resize, +Sort)

The user can modify the value of any cell when double clicking on a specific row.
After completing the modification, an editend event occurs. If false is returned at this time, the modified value is not applied.

No Name Age Location 

table_19 = table("#table_19", {
	fields: [ null, "name", "age", "location" ],
	data: table_data_big,
	editRow: true,
	resize: true,
	sort: true,
	event: {
		editend: function(data) {
			for(var key in data) {
				if(key == "age" && isNaN(data[key])) {
					alert("Enter a numerical value for the age.");
					return false;
				}
			}
		}
	},
    animate: true
});

Realtime Binding (1)

1sec 3sec 5sec 7sec 10sec
0 0 0 0 0

Input a data-bind property value in the tag to be bound as follows. The first value is the binding function name and the second value is the key value. 

<tr>
	<td data-bind="sec:1">0</td>
	<td data-bind="sec:3">0</td>
	<td data-bind="sec:5">0</td>
	<td data-bind="sec:7">0</td>
	<td data-bind="sec:10">0</td>
</tr>

If it is necessary to bind data dynamically, calling a function as follows bind.sec(key, value); 

table_7 = table("#table_7", {
	vo: true,
	animate: true
});
	
table7Sec(table_7.bind, 1);
table7Sec(table_7.bind, 3);
table7Sec(table_7.bind, 5);
table7Sec(table_7.bind, 7);
table7Sec(table_7.bind, 10);

function table7Sec(bind, sec) {
	var nowSecObj = bind.val("sec");
	
	setTimeout(function() {
		bind.sec(sec, parseInt(nowSecObj[sec]) + 1);
		table7Sec(bind, sec);
	}, sec * 1000);
}

function table8Color(bind, key) {
	var r = Math.floor(Math.random() * 255),
		g = Math.floor(Math.random() * 255),
		b = Math.floor(Math.random() * 255);
	
	var style = "rgb(" + r + "," + g + "," + b + ")";
	
	bind.color(key, style);
	bind.rgb(key, style);
}

Append/Insert Tree

Run - It is possible to input and add data for a specific index as a key value in a depth structure.

Index Column. Name Column. Age

How to create a tree table is the same as how to create an existing table. 

jui.ready([ "util.base", "ui.dropdown", "uix.table" ], function(_, dropdown, table) {
	var table_data_big = [
		{ name: "Hong", age: "20", location: "Ilsan" },
		{ name: "Jung", age: "30", location: "Seoul" },
		{ name: "Park", age: "15", location: "Yeosu" },
		{ name: "Kang", age: "32", location: "Seoul" },
		{ name: "Song", age: "12", location: "Gwangju" },
		{ name: "Yoon", age: "22", location: "Damyang" },
		{ name: "Kim", age: "33", location: "Busan" },
		{ name: "Hwang", age: "21", location: "Seoul" }
	];
		
	table_tree_1 = table("#table_tree_1", {
		data: table_data_big,
		scroll: true,
		resize: true,
		event: {
			select: function(row) {
				alert("index(" + row.index + "), name(" + row.data.name + ")");
			}
		}
	});
});

The append method adds a row as a child of a specified key value and the insert method adds a row at a location in the same line as the specified key value. 

function tableTree1Run() {
	table_tree_1.append("1", { name: "Kang", age: "21" });
	table_tree_1.append("1", { name: "Jung", age: "33" });
	table_tree_1.insert("1.2", { name: "Park", age: "45" } );
	table_tree_1.insert("1.3", { name: "Hwang", age: "12" } );
	table_tree_1.insert("1.2.0", { name: "Roo", age: "32" } );
	table_tree_1.insert("1.2.1", { name: "Jung", age: "14" } );
	table_tree_1.insert("3.0", { name: "Yoon", age: "17" } );
	table_tree_1.insert("3.1", { name: "Kim", age: "21" } );
	table_tree_1.insert("3.2", { name: "Kim", age: "28" } );
	
	table_tree_1.scroll();
}
