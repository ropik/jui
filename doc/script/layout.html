각각의 영역은 기본/최소/최대 크기를 정할 수 있는 옵션과 리사이징 옵션을 설정할 수 있습니다. 

jui.ready([ "ui.layout" ], function(layout) {
	layout("#container", {
		width: "auto",
		height: 300,
		left: {
			size: 100,
			min: 100,
			max: 300,
			resize: true
		},
		right: {
			size: 100,
			min: 100,
			max: 300,
			resize: true
		}
	});
});
<div id="container">
	<div class="left"><div class="title">Left</div></div>
	<div class="center"><div class="title">Center</div></div>
	<div class="right"><div class="title">Right</div></div>
</div>
