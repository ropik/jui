          jui.ready([ "uix.tree" ], function(tree) {
	tree_1 = tree(".tree", {
		root: { title: "C:\\" },
		event: {
			select: function(node) {

		}

function addTreeData(tree) {
	tree.append({ title: "Windows" });
	tree.append({ title: "Download" });
	tree.append({ title: "Program Files" });
	tree.append({ title: "Apache" });
	tree.append("0", { title: "run.exe" });
	tree.append("0", { title: "setting.conf" });
	tree.append("1", { title: "jui.torrrent" });
	tree.insert("2.0", { title: "Riot Games" });
	tree.insert("2.0.0", { title: "lol.exe" });
	tree.append("3", { title: "startup.bat" });
	tree.fold("0");
	tree.fold("1");
	tree.fold("3");
}

for(var i = 0; i < tree_1.length; i++) {
	addTreeData(tree_1[i]);
}

          Move/Remove/Update Node

          Move Remove Update - 노드를 조작할 수 있은 기능을 제공합니다. 

            function tree2RunA() {
	alert("Apache 폴더를 Program File 폴더로 이동합니다.");
	tree_1[4].move("3", "2.1");
}

function tree2RunB() {
	alert("Download 폴더를 삭제합니다.");
	tree_1[4].remove("1");
}
			
function tree2RunC() {
	alert("Program Files 폴더의 이름을 변경합니다.");
	tree_1[4].update("2", { title: "Applications" });
}

            Move Node (+Drag)

            각각의 노드를 마우스 드래그로 이동할 수 있는 기능을 제공합니다. 

              tree_2 = tree("#tree_2", {
	root: { title: "C:\\" },
	drag: true,
	tpl: {
		node: $("#tpl_tree").html()
	}
});

              Move Node (+Drag Child)

              dragChild 옵션의 값을 false로 설정하면, 드래그 대상 노드가 상대 노드의 자식으로 변경되는 것을 막을 수 있습니다. 

                tree_3 = tree("#tree_3", {
	root: { title: "C:\\" },
	drag: true,
	dragChild: false,
	tpl: {
		node: $("#tpl_tree").html()
	}
});
